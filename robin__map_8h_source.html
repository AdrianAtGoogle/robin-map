591  friend bool operator==(const robin_map& lhs, const robin_map& rhs) {
592  if(lhs.size() != rhs.size()) {
593  return false;
594  }
595 
596  for(const auto& element_lhs: lhs) {
597  const auto it_element_rhs = rhs.find(element_lhs.first);
598  if(it_element_rhs == rhs.cend() || element_lhs.second != it_element_rhs->second) {
599  return false;
600  }
601  }
602 
603  return true;
604  }
