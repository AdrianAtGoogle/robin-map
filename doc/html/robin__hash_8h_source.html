61 /**
62  * Grow the map by a factor of two keeping bucket_count to a power of two. It allows
63  * the map to use a mask operation instead of a modulo operation to map a hash to a bucket.
64  */
65 template<std::size_t GrowthFactor>
66 class power_of_two_growth_policy_rh {
67 public:
68  /**
69  * Called on map creation and rehash. The number of buckets requested is passed by parameter.
70  * This number is a minimum, the policy may update this value with a higher value if needed.
71  */
72  power_of_two_growth_policy_rh(std::size_t& min_bucket_count_in_out) {
73  if(min_bucket_count_in_out > max_bucket_count()) {
74  throw std::length_error("The map exceeds its maxmimum size.");
75  }
76 
77  static_assert(MIN_BUCKETS_SIZE > 0, "");
78  const std::size_t min_bucket_count = MIN_BUCKETS_SIZE;
79 
80  min_bucket_count_in_out = std::max(min_bucket_count, min_bucket_count_in_out);
81  min_bucket_count_in_out = round_up_to_power_of_two(min_bucket_count_in_out);
82  m_mask = min_bucket_count_in_out - 1;
83  }
84 
85  /**
86  * Return the bucket [0, bucket_count()) to which the hash belongs.
87  */
88  std::size_t bucket_for_hash(std::size_t hash) const {
89  return hash & m_mask;
90  }
91 
92  /**
93  * Return the bucket count to uses when the bucket array grows on rehash.
94  */
95  std::size_t next_bucket_count() const {
96  if((m_mask + 1) > max_bucket_count() / GrowthFactor) {
97  throw std::length_error("The map exceeds its maxmimum size.");
98  }
99 
100  return (m_mask + 1) * GrowthFactor;
101  }
102 
103  /**
104  * Return the maximum number of buckets supported by the policy.
105  */
106  std::size_t max_bucket_count() const {
107  return std::numeric_limits<std::size_t>::max() / GrowthFactor + 1;
108  }
109 
110 private:
111  static std::size_t round_up_to_power_of_two(std::size_t value) {
112  if(value == 0) {
113  return 1;
114  }
115 
116  if(is_power_of_two(value)) {
117  return value;
118  }
119 
120  --value;
121  for(std::size_t i = 1; i < sizeof(std::size_t) * CHAR_BIT; i *= 2) {
122  value |= value >> i;
123  }
124 
125  return value + 1;
126  }
127 
128  static constexpr bool is_power_of_two(std::size_t value) {
129  return value != 0 && (value & (value - 1)) == 0;
130  }
131 
132 private:
133  static const std::size_t MIN_BUCKETS_SIZE = 2;
134  static_assert(is_power_of_two(GrowthFactor), "GrowthFactor must be a power of two.");
135 
136  std::size_t m_mask;
137 };

139 /**
140  * Grow the map by GrowthFactor::num/GrowthFactor::den and use a modulo to map a hash
141  * to a bucket. Slower but it can be usefull if you want a slower growth.
142  */
143 template<class GrowthFactor = std::ratio<3, 2>>
144 class mod_growth_policy_rh {
145 public:
146  mod_growth_policy_rh(std::size_t& min_bucket_count_in_out) {
147  if(min_bucket_count_in_out > max_bucket_count()) {
148  throw std::length_error("The map exceeds its maxmimum size.");
149  }
150 
151  static_assert(MIN_BUCKETS_SIZE > 0, "");
152  const std::size_t min_bucket_count = MIN_BUCKETS_SIZE;
153 
154  min_bucket_count_in_out = std::max(min_bucket_count, min_bucket_count_in_out);
155  m_bucket_count = min_bucket_count_in_out;
156  }
157 
158  std::size_t bucket_for_hash(std::size_t hash) const {
159  tsl_assert(m_bucket_count != 0);
160  return hash % m_bucket_count;
161  }
162 
163  std::size_t next_bucket_count() const {
164  if(m_bucket_count == max_bucket_count()) {
165  throw std::length_error("The map exceeds its maxmimum size.");
166  }
167 
168  const double next_bucket_count = std::ceil(double(m_bucket_count) * REHASH_SIZE_MULTIPLICATION_FACTOR);
169  if(!std::isnormal(next_bucket_count)) {
170  throw std::length_error("The map exceeds its maxmimum size.");
171  }
172 
173  if(next_bucket_count > double(max_bucket_count())) {
174  return max_bucket_count();
175  }
176  else {
177  return std::size_t(next_bucket_count);
178  }
179  }
180 
181  std::size_t max_bucket_count() const {
182  return MAX_BUCKET_COUNT;
183  }
184 
185 private:
186  static const std::size_t MIN_BUCKETS_SIZE = 2;
187  static constexpr double REHASH_SIZE_MULTIPLICATION_FACTOR = 1.0*GrowthFactor::num/GrowthFactor::den;
188  static const std::size_t MAX_BUCKET_COUNT =
189  std::size_t(double(
190  std::numeric_limits<std::size_t>::max()/REHASH_SIZE_MULTIPLICATION_FACTOR
191  ));
192 
193  static_assert(REHASH_SIZE_MULTIPLICATION_FACTOR >= 1.1, "Growth factor should be >= 1.1.");
194 
195  std::size_t m_bucket_count;
196 };

223 /**
224  * Grow the map by using prime numbers as size. Slower than tsl::power_of_two_growth_policy_rh in general
225  * but will probably distribute the values around better in the buckets with a poor hash function.
226  */
227 class prime_growth_policy_rh {
228 public:
229  prime_growth_policy_rh(std::size_t& min_bucket_count_in_out) {
230  auto it_prime = std::lower_bound(tsl::detail_robin_hash::PRIMES.begin(),
231  tsl::detail_robin_hash::PRIMES.end(), min_bucket_count_in_out);
232  if(it_prime == tsl::detail_robin_hash::PRIMES.end()) {
233  throw std::length_error("The map exceeds its maxmimum size.");
234  }
235 
236  m_iprime = static_cast<unsigned int>(std::distance(tsl::detail_robin_hash::PRIMES.begin(), it_prime));
237  min_bucket_count_in_out = *it_prime;
238  }
239 
240  std::size_t bucket_for_hash(std::size_t hash) const {
241  return bucket_for_hash_iprime(hash, m_iprime);
242  }
243 
244  std::size_t next_bucket_count() const {
245  if(m_iprime + 1 >= tsl::detail_robin_hash::PRIMES.size()) {
246  throw std::length_error("The map exceeds its maxmimum size.");
247  }
248 
249  return tsl::detail_robin_hash::PRIMES[m_iprime + 1];
250  }
251 
252  std::size_t max_bucket_count() const {
253  return tsl::detail_robin_hash::PRIMES.back();
254  }
255 
256 private:
257  std::size_t bucket_for_hash_iprime(std::size_t hash, unsigned int iprime) const {
258  tsl_assert(iprime < tsl::detail_robin_hash::MOD_PRIME.size());
259  return tsl::detail_robin_hash::MOD_PRIME[iprime](hash);
260  }
261 
262 private:
263  unsigned int m_iprime;
264 };
